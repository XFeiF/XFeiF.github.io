def minFallingPathSum(self, A):
    if not A:
        return 0
    if len(A) == 1:
        return min(A[-1])
    size = len(A)
    res = [[0]*size for i in range(size)]
    res[0] = A[0]
    for row in range(1, size):
        for col in range(size):
            if col == 0:
                res[row][col] = A[row][col] + min(res[row-1][col], res[row-1][col+1])
            elif col == size - 1:
                res[row][col] = A[row][col] + min(res[row-1][col-1], res[row-1][col])
            else:
                res[row][col] = A[row][col] + min(res[row-1][col-1], res[row-1][col], res[row-1][col+1])
    return min(res[-1])

def canIWin(self, M, T):
    def win(M, T, m, state):
        if T <= 0: return False
        if m[state] != 0: return m[state] == 1

            return True
        m[state] = -1
        return False

    if T <= 0: return True
    if s == T: return (M % 2) == 1

    i = 0

def findLongestChain(self, pairs):

def maximalSquare(self, matrix):

    n = len(matrix)
    m = len(matrix[0])

    res = 0

def maxProduct(self, nums):
    n = len(nums)
    res = nums[0]
    rx, rm = nums[0], nums[0]
    for i in range(1, n):
        if nums[i] < 0:
            tmp = rx
            rx = rm
            rm = tmp
        rx = max(nums[i], nums[i]*rx)
        rm = min(nums[i], nums[i]*rm)
        res = max(res, rx)
    return res

def largestSumOfAverages(self, A, K):
    memo = {}
    def search(n, k):
        if (n, k) in memo: return memo[n, k]
        if n < k: return 0
        if k == 1:
            memo[n, k] = sum(A[:n]) / float(n)
            return memo[n, k]
        cur, memo[n, k] = 0, 0

            cur += A[i]
            memo[n, k] = max(memo[n, k], search(i, k - 1) + cur / float(n - i))
        return memo[n, k]
