    def minPathSum(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n = len(grid)
        m = len(grid[0])

        for i in range(1,m):
            dp[0][i] = dp[0][i-1] + grid[0][i]

class Solution:
    def integerBreak(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n < 4:
            return n-1
        maxr = [0, 1, 2, 3]
        for cur in range(4, n+1):
            cm = 0
            for i in range(1, int(cur/2) + 1):

        """
        :type s: str
        :type t: str
        :rtype: bool
        """
        def sub(s, t):

            if not t: return False

                if target == t[i]:
                    idx = i
                    break

            return sub(s[1:], t[idx+1:])
        return sub(s, t)
