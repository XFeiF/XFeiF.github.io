def maxProfit(self, prices):
    if len(prices) < 2: return 0
    sell, buy, prev_sell, prev_buy = 0, -prices[0], 0, 0
    for price in prices:
        prev_buy = buy
        buy = max(prev_sell - price, prev_buy)
        prev_sell = sell
        sell = max(prev_buy + price, prev_sell)
    return sell

def orderOfLargestPlusSign(self, N, mines):

    dp = [[0] * (N+2) for _ in range(N+2)]
    up = [[0] * (N+2) for _ in range(N+2)]
    down = [[0] * (N+2) for _ in range(N+2)]
    left = [[0] * (N+2) for _ in range(N+2)]
    right = [[0] * (N+2) for _ in range(N+2)]
    for i in range(1, N+1):
        for j in range(1, N+1):
            dp[i][j] = 1

        x, y = mine
        matrix[x][y] = 0
        dp[x+1][y+1] = 0

        for j in range(1, N+1):
            if matrix[i-1][j-1]:
                up[i][j] = up[i-1][j] + matrix[i-1][j-1]

def minFallingPathSum(self, A):
    if not A:
        return 0
    if len(A) == 1:
        return min(A[-1])

    res = [[0]*size for i in range(size)]

def canIWin(self, M, T):
    def win(M, T, m, state):
        if T <= 0: return False
        if m[state] != 0: return m[state] == 1
        for i in range(M):
        if (state & (1 << i)) > 0: continue
        if not win(M, T - i - 1, m, state | (1 << i)):
            m[state] = 1
            return True
        m[state] = -1
        return False

    s = M * (M + 1) / 2
    if s < T: return False
    if T <= 0: return True
    if s == T: return (M % 2) == 1

    m = [0] * (1 << M)
    return win(M, T, m, 0)

def pushDominoes(self, d):
    d = 'L' + d + 'R'
    res = []
    i = 0
    for j in range(1, len(d)):

def findLongestChain(self, pairs):
    pairs.sort(key=lambda r:r[0])
    n = len(pairs)
    dp = [0]*n
    dp[0] = 1
    if pairs[1][0] > pairs[0][1]:
        dp[1] = dp[0] + 1
    else:
        dp[1] = 1

def maxProduct(self, nums):
    n = len(nums)
    res = nums[0]
    rx, rm = nums[0], nums[0]
    for i in range(1, n):
        if nums[i] < 0:
            tmp = rx
            rx = rm
            rm = tmp
        rx = max(nums[i], nums[i]*rx)
        rm = min(nums[i], nums[i]*rm)
        res = max(res, rx)
    return res

    def minPathSum(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n = len(grid)
        m = len(grid[0])
        dp = [[0]*m for _ in range(n)]
        dp[0][0] = grid[0][0]
        for i in range(1,m):
            dp[0][i] = dp[0][i-1] + grid[0][i]

        :type nums: List[int]
        :rtype: int

        """
        :type n: int
        :rtype: int
        """
        if n < 4:
            return n-1
        maxr = [0, 1, 2, 3]
        for cur in range(4, n+1):

    def isSubsequence(self, s, t):
        """
        :type s: str
        :type t: str
        :rtype: bool
        """
        def sub(s, t):
            if not s: return True
            if not t: return False
            target = s[0]
            idx = -1
            for i in range(len(t)):
                if target == t[i]:
                    idx = i

            return sub(s[1:], t[idx+1:])

            for i in range(n):
                j = i+l

    maxPrice = T[0]
    for i in range(1, n):
        if prices[i] < prices[i-1]:
            T.append(T[i-1])
            minPrice = prices[i] if prices[i] < minPrice else minPrice
        else:
            cur = prices[i] - minPrice
            maxPrice = cur if cur > maxPrice else maxPrice
            T.append(maxPrice)
    return T[n-1]

class Solution:
    def climbStairs(self, n):
        """
        :type n: int
        :rtype: int
        """
        lookup = [None]*(n+1)

        # Memoization -> Bottom Up
        def fib(n):
            if n == 0 or n == 1:
                lookup[n] = 1
            if lookup[n] is None:
                lookup[n] = fib(n-1) + fib(n-2)
            return lookup[n]
        res = fib(n)
        # print(lookup)
        return res
